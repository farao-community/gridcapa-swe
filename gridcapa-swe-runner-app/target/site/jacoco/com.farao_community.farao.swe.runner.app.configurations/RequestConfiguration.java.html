package com.farao_community.farao.swe.runner.app.configurations;

import com.farao_community.farao.swe.runner.app.services.RequestService;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import reactor.core.publisher.Flux;

import java.util.function.Function;

/**
 * @author Theo Pascoli {@literal <theo.pascoli at rte-france.com>}
 */
@Configuration
public class RequestConfiguration {

    @Bean
    public Function<Flux<byte[]>, Flux<byte[]>> request(RequestService requestService) {
        return sweRequestFlux -> sweRequestFlux
                .map(requestService::launchSweRequest)
                .log();
    }

}
